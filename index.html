        function setLanguage(lang) {

            currentLanguage = lang;

            const langData = translations[lang];


            // Establecer el idioma de la página y el título del documento

            document.documentElement.lang = lang;

            document.title = langData['page-title'];


            // Actualizar todo el texto con data-key

            document.querySelectorAll('[data-key]').forEach(el => {

                const key = el.dataset.key;

                if (langData[key]) {

                    // Usar innerHTML para elementos que contienen iconos

                    if (key.startsWith('btn-') || key.startsWith('footer-link-')) {

                        el.innerHTML = langData[key];

                    } else {

                        el.innerText = langData[key];

                    }

                }

            });


            // Actualizar el texto del botón de cambio de idioma

            const toggleBtn = document.getElementById('lang-toggle-btn');

            if (lang === 'en') {

                toggleBtn.innerText = 'Ver en Español';

            } else {

                toggleBtn.innerText = 'View in English';

            }

        }


        function toggleLanguage() {

            const newLang = currentLanguage === 'en' ? 'es' : 'en';

            setLanguage(newLang);

        }
